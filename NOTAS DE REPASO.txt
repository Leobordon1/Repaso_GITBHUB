def calcular_promedio(lista):
    promedio = sum(lista) / len(lista)
    mayor = max(lista)
    menor = min(lista)
    return {
        "Promedio": promedio,
        "Mayor": mayor,
        "Menor": menor
    }


numeros = [5,10,15]
diccionario = calcular_promedio(numeros)
print("Promedio:", diccionario["Promedio"])
print("Mayor:", diccionario["Mayor"])
print("Menor:", diccionario["Menor"])
